        // Generate Random Key
        function generateKey() {
            const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/';
            let key = '';
            for (let i = 0; i < 44; i++) {
                key += chars.charAt(Math.floor(Math.random() * chars.length));
            }
            return key + '=';
        }

        // Generate QR Code
        function generateQR(text) {
            const canvas = document.getElementById('qr-canvas');
            const ctx = canvas.getContext('2d');
            ctx.fillStyle = 'white';
            ctx.fillRect(0, 0, 200, 200);
            ctx.fillStyle = '#0a0a0f';
            
            const cellSize = 5;
            const gridSize = 40;
            
            let seed = 0;
            for (let i = 0; i < text.length; i++) seed += text.charCodeAt(i);
            
            for (let x = 0; x < gridSize; x++) {
                for (let y = 0; y < gridSize; y++) {
                    const pseudoRandom = Math.sin(seed * x * y * 0.1) * 10000;
                    if ((pseudoRandom - Math.floor(pseudoRandom)) > 0.5) {
                        ctx.fillRect(x * cellSize, y * cellSize, cellSize, cellSize);
                    }
                }
            }
            
            ctx.fillStyle = '#0a0a0f';
            ctx.fillRect(0, 0, 35, 35);
            ctx.clearRect(5, 5, 25, 25);
            ctx.fillRect(10, 10, 15, 15);
            
            ctx.fillRect(165, 0, 35, 35);
            ctx.clearRect(170, 5, 25, 25);
            ctx.fillRect(175, 10, 15, 15);
            
            ctx.fillRect(0, 165, 35, 35);
            ctx.clearRect(5, 170, 25, 25);
            ctx.fillRect(10, 175, 15, 15);
        }
